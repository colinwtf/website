import java.util.Scanner;

public class Main
{
   public static void main(String[] args)
   {
      Scanner scanner = new Scanner(System.in);
      
      // Prompt the user to enter the first integer
      System.out.println("Enter an integer:");
      int n1 = scanner.nextInt();
      
      // Prompt the user to enter the second integer
      System.out.println("Enter another integer:");
      int n2 = scanner.nextInt();
      
      // Calculate the sum and product
      int sum = n1 + n2;
      int product = n1 * n2;

public class StringManipulation
{
   
   public static String removeCharacters(String input)
   {
      if (input.length() >= 8)
      {
         // Create a new string without the 3rd and 8th characters
         return input.substring(0, 2) + input.substring(3, 7) + input.substring(8);
      }
      else if (input.length() >= 3)
      {
         // If the string is at least 3 characters long but less than 8, only remove the 3rd character
         return input.substring(0, 2) + input.substring(3);
      }
      else
      {
         // If the string is shorter than 3 characters, return it as is
         return input;
      }
   }
   
   public static void main(String[] args)
   {
      String str1 = "abcdefghij";
      String result1 = removeCharacters(str1);
      System.out.println("Result for '"+ str1 + "': "+ result1);
      
      String str2 = "abcde";
      String result2 = removeCharacters(str2);
      System.out.println("Result for '"+ str2 + "': "+ result2);
      
      String str3 = "ab";
      String result3 = removeCharacters(str3);
      System.out.println("Result for '"+ str3 + "': "+ result3);
   }
}
